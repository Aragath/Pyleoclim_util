Examples

In [1]: import pyleoclim as pyleo

In [2]: import pandas as pd

In [3]: from matplotlib import pyplot as plt

In [4]: data=pd.read_csv('https://raw.githubusercontent.com/LinkedEarth/Pyleoclim_util/Development/example_data/wtc_test_data_nino.csv')

In [5]: t=data.iloc[:,0]

In [6]: air=data.iloc[:,1]

In [7]: nino=data.iloc[:,2]

In [8]: ts_nino=pyleo.Series(time=t,value=nino)

In [9]: ts_air=pyleo.Series(time=t,value=air)

#plot the two timeseries
In [10]: fig, ax = ts_nino.plot(title='El Nino Region 3 -- SST Anomalies')
<Figure size 1000x400 with 1 Axes>

In [11]: plt.close(fig)

In [12]: fig, ax = ts_air.plot(title='Deasonalized All Indian Rainfall Index')
<Figure size 1000x400 with 1 Axes>

In [13]: plt.close(fig)

In [14]: ts_air_std=ts_air.standardize()

In [15]: ts_nino_std=ts_nino.standardize()

In [16]: coh = ts_nino.wavelet_coherence(ts_air)

In [17]: coh_signif = coh.signif_test(number=10, qs=[0.99])

In [18]: fig, ax = coh_signif.plot(phase_style={'skip_x': 50, 'skip_y': 10})
<Figure size 1000x800 with 2 Axes>

In [19]: plt.close(fig)
